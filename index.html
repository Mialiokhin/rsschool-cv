Who is the killer? KATA from Codewars:
Given a dictionary with all the names of the suspects and everyone that they have seen on that day and also a list of the names of the dead people. You need to return the name of the one killer.

function killer(suspectInfo, dead) {
let Killer = ''
let allKills = 0
for ([key,value] of Object.entries(suspectInfo)){
let count = 0
dead.forEach(x => {
if (value.includes(x)) {
count+=1
}
if (count > allKills) {
allKills = count
Killer = key
}
})
}
return Killer
}
